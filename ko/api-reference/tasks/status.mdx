```c C
#include <stdio.h>
#include <curl/curl.h>

int main(void) {
    CURL *curl;
    CURLcode res;

    curl_global_init(CURL_GLOBAL_DEFAULT);
    curl = curl_easy_init();

    if(curl) {
        const char *url = "https://api.apimart.ai/v1/tasks/task-unified-1757156493-imcg5zqt";

        struct curl_slist *headers = NULL;
        headers = curl_slist_append(headers, "Authorization: Bearer <token>");

        curl_easy_setopt(curl, CURLOPT_URL, url);
        curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);

        res = curl_easy_perform(curl);

        if(res != CURLE_OK) {
            fprintf(stderr, "curl_easy_perform() failed: %s\n",
                    curl_easy_strerror(res));
        }

        curl_slist_free_all(headers);
        curl_easy_cleanup(curl);
    }

    curl_global_cleanup();
    return 0;
}
```

## 인증

<ParamField header="Authorization" type="string" required>
  모든 API 엔드포인트에는 Bearer Token 인증이 필요합니다

  API 키 가져오기:

  [API 키 관리 페이지](https://api.apimart.ai/console/token)를 방문하여 API 키를 가져오세요

  요청 헤더에 추가:

  ```
  Authorization: Bearer YOUR_API_KEY
  ```
</ParamField>

## 경로 매개변수

<ParamField path="task_id" type="string" required>
  생성 API에서 반환된 작업 ID
</ParamField>

## 응답

<ResponseField name="id" type="string">
  고유한 작업 식별자
</ResponseField>

<ResponseField name="status" type="string">
  작업 상태 값:
  - `pending` - 처리 대기 중
  - `processing` - 진행 중
  - `completed` - 성공적으로 완료됨
  - `failed` - 실패
  - `cancelled` - 사용자에 의해 취소됨
</ResponseField>

<ResponseField name="progress" type="integer">
  작업 진행률 (0–100)
</ResponseField>

<ResponseField name="result" type="object">
  작업 결과, 상태가 `completed`일 때만 반환됩니다

  <Expandable title="속성">
    <ResponseField name="images" type="array">
      생성된 이미지 객체 배열 (이미지 생성 작업의 경우)
    </ResponseField>

    <ResponseField name="videos" type="array">
      생성된 비디오 객체 배열 (비디오 생성 작업의 경우)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="created" type="integer">
  작업 생성 타임스탬프
</ResponseField>

<ResponseField name="completed" type="integer">
  작업 완료 타임스탬프 (완료 시에만 존재)
</ResponseField>

<ResponseField name="estimated_time" type="integer">
  예상 완료 시간(초)
</ResponseField>

<ResponseField name="actual_time" type="integer">
  실제 완료 시간(초) (완료 시에만 존재)
</ResponseField>

<ResponseField name="error" type="object">
  오류 세부정보 (상태가 `failed`일 때만 존재)

  <Expandable title="속성">
    <ResponseField name="code" type="integer">
      오류 코드
    </ResponseField>

    <ResponseField name="message" type="string">
      오류 메시지
    </ResponseField>

    <ResponseField name="type" type="string">
      오류 유형
    </ResponseField>
  </Expandable>
</ResponseField>
