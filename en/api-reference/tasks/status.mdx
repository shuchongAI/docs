```c C
#include <stdio.h>
#include <curl/curl.h>

int main(void) {
    CURL *curl;
    CURLcode res;

    curl_global_init(CURL_GLOBAL_DEFAULT);
    curl = curl_easy_init();

    if(curl) {
        const char *url = "https://api.apimart.ai/v1/tasks/task-unified-1757156493-imcg5zqt?language=en";

        struct curl_slist *headers = NULL;
        headers = curl_slist_append(headers, "Authorization: Bearer <token>");

        curl_easy_setopt(curl, CURLOPT_URL, url);
        curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);

        res = curl_easy_perform(curl);

        if(res != CURLE_OK) {
            fprintf(stderr, "curl_easy_perform() failed: %s\n",
                    curl_easy_strerror(res));
        }

        curl_slist_free_all(headers);
        curl_easy_cleanup(curl);
    }

    curl_global_cleanup();
    return 0;
}
```

## Authorizations

<ParamField header="Authorization" type="string" required>
  All API endpoints require Bearer Token authentication

  Get your API Key:

  Visit the [API Key Management Page](https://api.apimart.ai/console/token) to get your API Key

  Add it to the request header:

  ```
  Authorization: Bearer YOUR_API_KEY
  ```
</ParamField>

## Path parameters

<ParamField path="task_id" type="string" required>
  Task ID returned by the generation API
</ParamField>

## Query parameters

<ParamField query="language" type="string">
  Language for response content, supports the following values:
  - `zh` - Chinese
  - `en` - English
  - `ko` - Korean
  - `ja` - Japanese
  
  Defaults to English
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique task identifier
</ResponseField>

<ResponseField name="status" type="string">
  Task status values:
  - `pending` - Queued for processing
  - `processing` - In progress
  - `completed` - Finished successfully
  - `failed` - Failed
  - `cancelled` - Cancelled by user
</ResponseField>

<ResponseField name="progress" type="integer">
  Task progress percentage (0–100)
</ResponseField>

<ResponseField name="result" type="object">
  Task result, returned only when status is `completed`

  <Expandable title="Properties">
    <ResponseField name="images" type="array">
      Array of generated image objects (for image generation tasks)
    </ResponseField>

    <ResponseField name="videos" type="array">
      Array of generated video objects (for video generation tasks)
    </ResponseField>

    <ResponseField name="thumbnail_url" type="string">
      Video thumbnail URL (for video generation tasks)
      
      <Note>
      **Important:** This field is only returned when thumbnail generation is explicitly specified in the request parameters when creating the task. If thumbnail generation is not requested, this field will not be included in the response.
      </Note>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="created" type="integer">
  Task creation timestamp
</ResponseField>

<ResponseField name="completed" type="integer">
  Task completion timestamp (only present when completed)
</ResponseField>

<ResponseField name="estimated_time" type="integer">
  Estimated completion time in seconds
</ResponseField>

<ResponseField name="actual_time" type="integer">
  Actual completion time in seconds (only present when completed)
</ResponseField>

<ResponseField name="error" type="object">
  Error details (only present when status is `failed`)

  <Expandable title="Properties">
    <ResponseField name="code" type="integer">
      Error code
    </ResponseField>

    <ResponseField name="message" type="string">
      Error message
    </ResponseField>

    <ResponseField name="type" type="string">
      Error type
    </ResponseField>
  </Expandable>
</ResponseField>
