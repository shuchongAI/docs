import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.URI;

public class Main {
    public static void main(String[] args) throws Exception {
        String url = "https://api.apimart.ai/v1/images/generations";

        String payload = """
        {

        }
        """;

        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(url))
            .header("Authorization", "Bearer <token>")
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(payload))
            .build();

        HttpResponse<String> response = client.send(request,
            HttpResponse.BodyHandlers.ofString());

        System.out.println(response.body());
    }

## Body

<ParamField body="model" type="string" default="gemini-3.1-flash-image-preview" required>
  画像生成モデル名

  Example: `"gemini-3.1-flash-image-preview"`
</ParamField>

<ParamField body="prompt" type="string" required>
  画像生成のテキスト説明
</ParamField>

<ParamField body="size" type="string">
  画像のアスペクト比

  サポートされる比率：
  - `1:1` - 正方形、アバター、SNS
  - `3:2` / `2:3` - 標準写真
  - `4:3` / `3:4` - 従来のディスプレイ比率
  - `16:9` / `9:16` - ワイドスクリーン/縦型動画カバー
  - `5:4` / `4:5` - Instagram画像
  - `21:9` - ウルトラワイドバナー
  - `1:4` / `4:1` - 長尺ポスター/バナー
  - `1:8` / `8:1` - 極端な長尺画像/バナー広告
</ParamField>

<ParamField body="resolution" type="string" default="1K">
  出力画像解像度

  サポートされる値：
  - `0.5K` - 約512px、低解像度プレビュー
  - `1K` - 約1024px、標準解像度（デフォルト）
  - `2K` - 約2048px、高解像度
  - `4K` - 約4096px、超高解像度

  > **注意：** 解像度によって料金が異なります。4Kは1Kより高額です。
</ParamField>

<ParamField body="n" type="integer">
  生成する画像の数

  デフォルト：1

  **⚠️ 注意：** 数値のみ入力してください（例：`1`）。引用符を使用するとエラーが発生します
</ParamField>

<ParamField body="image_urls" type="array">
  画像から画像生成用の参照画像URLリスト

  **制限：**
  - 最大 **14枚** の参照画像（推奨：最大10枚のオブジェクト参照 + 4枚のキャラクター参照）
  - 1枚あたり10MB以下
  - 対応形式：jpeg、png、webp
</ParamField>

<ParamField body="google_search" type="boolean" default="false">
  Googleテキスト検索強化を有効にする

  - `true`：モデルがウェブテキスト情報を検索して画像生成を支援します
  - `false`：無効（デフォルト）

  > **注意：** `google_search` は `tools` パラメータと排他的です
</ParamField>

<ParamField body="google_image_search" type="boolean" default="false">
  Google画像検索強化を有効にする

  - `true`：テキスト検索に加えて、参照画像も検索して生成を支援します
  - `false`：無効（デフォルト）

  > **注意：** `google_search: true` と一緒に使用する必要があります
</ParamField>
